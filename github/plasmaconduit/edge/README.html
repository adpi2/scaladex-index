Edge

An asynchronous http server and client implementation built on top of Netty and Conveyance.

Installation

Add the following to your Build.scala

resolvers += "Plasma Conduit Repository" at "http://dl.bintray.com/plasmaconduit/releases",
libraryDependencies += "com.plasmaconduit" %% "edge" % "0.38.0"

Usage

package com.plasmaconduit.edge

import com.plasmaconduit.edge.http.{Ok, InternalServerError}
import com.plasmaconduit.url.URL

object Main {

  def main(args: Array[String]): Unit = {
    HttpServer(9999, { request =>
      HttpClient.get(URL("http://ip-api.com/json").get)
        .mapError(_.throwable)
        .map({ ipResponse => Ok(ipResponse.content.retain()) })
        .onErrorReturn(InternalServerError("Failed contacting ip service"))
    }).startAndWait
  }

}
